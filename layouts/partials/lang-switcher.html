{{- /* Custom 4-language switcher. - 中 (zh) → current page (defaultContentLanguageInSubdir=false so zh has no prefix) - EN / 越 → real translated pages (construct URL manually from RelPermalink + lang prefix) - 日 (ja) → Google Translate URL wrapper (no real translation yet) Note: Hugo v0.161 has a quirk where .Translations on the default-lang page returns empty, so we build the EN/VI URLs manually from the current page's relative path. */ -}} {{- $currentUrl := .Permalink -}} {{- $currentRel := .RelPermalink -}} {{- $parsed := urls.Parse .Permalink -}} {{- $host := $parsed.Host -}} {{- /* detect current lang from URL */ -}} {{- $currentLang := "zh" -}} {{- if strings.Contains $currentRel "/en/" }}{{ $currentLang = "en" }}{{ end -}} {{- if strings.Contains $currentRel "/vi/" }}{{ $currentLang = "vi" }}{{ end -}} {{- if strings.Contains $currentRel "/ja/" }}{{ $currentLang = "ja" }}{{ end -}} {{- /* build EN/VI URLs: base + lang prefix + path-without-lang-prefix */ -}} {{- $cleanRel := $currentRel -}} {{- if strings.Contains $currentRel "/en/" }}{{ $cleanRel = replace $cleanRel "/en/" "/" 1 }}{{ end -}} {{- if strings.Contains $currentRel "/vi/" }}{{ $cleanRel = replace $cleanRel "/vi/" "/" 1 }}{{ end -}} {{- if strings.Contains $currentRel "/ja/" }}{{ $cleanRel = replace $cleanRel "/ja/" "/" 1 }}{{ end -}} {{- $base := printf "https://%s" $host -}} {{- $enUrl := printf "%s%s" $base (replace $cleanRel "/posts/" "/en/posts/" 1) -}} {{- $viUrl := printf "%s%s" $base (replace $cleanRel "/posts/" "/vi/posts/" 1) -}} {{- $jaUrl := printf "https://translate.google.com/translate?sl=%s&tl=ja&u=%s" $currentLang ($currentUrl | safeURL) -}}
